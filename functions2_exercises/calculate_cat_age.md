### Solution 
``` JavaScript 
let catAge = function (humanAge) {
        if (humanAge === 0) {
                console.log(0);
        }else if (humanAge === 1) {
                console.log(15);
        }else if (humanAge === 2) {
                console.log(24);
        }else if (humanAge === 3) {
                console.log(28);
        }else {
                console.log(32);
        }
}



catAge(0); // 0
catAge(1); // 15
catAge(2); // 24
catAge(3); // 28
```
